BEGIN main
	SET currentStoreData to first item in CALL storeInit
	SET currentStore to second item in CALL storeInit
	CALL displayStore with data inputs of currentStore, currentStoreData
	ONTRIGGER CALL addnewItem with data inputs of currentStore,  currentStoreData
	ONTRIGGER CALL addnewCustomer with data inputs of currentStore,  currentStoreData
END main

BEGIN addnewCustomer 
	PROMPT customerName
	SET numOrders to 0
	SET selectingItems to TRUE
	IF customerName is in currentStoreData customer data THEN
		DISPLAY Customer already exists
		SET action to PROMPT "Do you want to modify(1)/remove(2) that customer's order or Cancel action(3)"
		IF action = modify or 1 THEN
			CALL modifyOrdercustomer with data input of customerName, currentStore, currentStoreData
		ELSE IF action = remove or 2 THEN
			CALL removeOrdercustomer with data input of customerName, currentStore, currentStoreData
		END IF
		RETURN
	END IF

	CREATE newCustomer WITH data
		name = customerName
		orders = empty array
	END CREATE 

	WHILE selectingItems = TRUE DO
		SET currentTotalPrice to 0
		CALL displayStore AND PROMPT selectItem
		DISPLAY price, description of selectItem
		PROMPT itemQty
		IF itemQty <= 0 THEN
			SET removeItemloop to TRUE
		END IF

		WHILE removeItemloop = TRUE  DO
			PROMPT removeItemFromOrder
			IF removeItemFromOrder = n THEN
				PROMPT itemQty
				IF itemQty > 0 THEN
					removeItemloop = FALSE
				END IF
			ELSE IF removeItemFromOrder = y THEN
				removeItemloop = FALSE
			END IF
		END WHILE
		IF removeItemFromOrder = n THEN
			SET orderName to "order" JOINING numOrderasdasdAds
			ADD orderName to newCustomer WITH data	
				item = selectItem
				Qty = itemQty
				price = price of selectItem
			END ADD
		END IF
		FOR order in newCustomer GET orders
			ADD price from order to currentTotalPrice
		DISPLAY newCustomer orders AND currentTotalPrice
		PROMPT continueSelectingItemes
		IF continueSelectingItemes = n THEN
			selectingItems = FALSE 
		END IF 
	END WHILE
	ADD newCustomer to currentStoreData customer data
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage
	CALL displayStore with data inputs of currentStore, and currentStoreData
END newCustomer

BEGIN modifyOrdercustomer
	SET customerData to customer data in currentStoreData
	SET orders to customerName in customerData orders data
	SET orderSelecting to TRUE
	SET numOrders to GET number of elements in orders

	WHILE orderSelecting = TRUE DO
		SET currentTotalPrice to 0

		FOR each order in orders
			SET choice to index of each order within orders
			DISPLAY choice+1, item from order, Qty from order, price from order
		END FOR

		PROMPT selectedOrder

		IF selectedOrder < numOrders+1 AND selectedOrder > 0 THEN
			SET selectedOrderData to orders[selectedOrder-1]

			SET editAction to PROMPT modify quantity(1)/remove item(2)/cancel(3)

			IF editAction = modify OR 1 THEN
				PROMPT newQty

				IF newQty <= 0 THEN
					DISPLAY Quantity must be greater than 0
				ELSE
					SET Qty in selectedOrderData to newQty
				END IF

			ELSE IF editAction = remove OR 2 THEN
				REMOVE selectedOrderData from orders

			ELSE IF editAction = cancel OR 3 THEN
				orderSelecting = FALSE
			END IF

			FOR each order in orders
				ADD (price from order * Qty from order) to currentTotalPrice
			END FOR

			DISPLAY updated orders
			DISPLAY currentTotalPrice

			PROMPT continueEditing

			IF continueEditing = n THEN
				orderSelecting = FALSE
			END IF
		END IF
	END WHILE

	SAVE updated orders to customerName in customerData
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage

	CALL displayStore with data inputs of currentStore, and currentStoreData
END modifyOrdercustomer

BEGIN removeOrdercustomer
	PROMPT confirmRemoval

	IF confirmRemoval = y THEN
		REMOVE customerName from currentStoreData customer data

		SAVE updated currentStoreData to stores
		SAVE updated stores to storage

		CALL displayStore with data inputs of currentStore, and currentStoreData
	END IF

	RETURN
END removeOrdercustomer

BEGIN addnewItem
	PROMPT itemName
	SET editing to TRUE
	IF itemName is in currentStoreData items data THEN
		DISPLAY Item already exists
		SET action to PROMPT want to modify(1)/remove(2) that item or Cancel action(3)
		IF action = modify or 1 THEN
			CALL modifyItem with data input of itemName, currentStore, currentStoreData
		ELSE IF action = remove or 2 THEN
			CALL removeItem with data input of itemName, currentStore, currentStoreData
		END IF
		RETURN
	END IF
	WHILE editing = TRUE DO
		PROMPT itemPrice
		PROMPT itemProductionCost
		PROMPT itemDescription
		PROMPT itemComplete
		IF itemComplete = y
			editing = FALSE
		END IF
	END WHILE
	CREATE newItem WITH data
		name = itemName
		price = itemPrice
		productionCost = itemProductionCost
		description = itemDescription
	END CREATE
	ADD newItem to currentStoreData items data
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage
	CALL displayStore with data inputs of currentStore, and currentStoreData
END addnewItem

BEGIN modifyItem
	SET itemData to GET itemName data from currentStoreData
	SET modifying to TRUE
	WHILE modifying = TRUE DO
		DISPLAY itemData
		PROMPT newItemPrice
		PROMPT newItemProductionCost
		PROMPT newItemDescription
		PROMPT modificationComplete
		IF modificationComplete = y
			modifying = FALSE
		END IF
	END WHILE
	SET itemName itemData to
		price = newItemPrice
		productionCost = newItemProductionCost
		description = newItemDescription
	END SET
	SAVE updated itemName to currentStoreData item data
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage
	CALL displayStore with data inputs of currentStore, and currentStoreData
END modifyItem

BEGIN removeItem
	PROMPT confirmRemoval
	IF confirmRemoval = y
		REMOVE itemName from currentStoreData
		SAVE updated currentStoreData to stores
		SAVE updated stores to storage
		CALL displayStore with data inputs of currentStore, and currentStoreData
	END IF
	RETURN
END removeItem

BEGIN displayStore
	DISPLAY restaurant name USING currentStore
	DISPLAY menu items USING currentStoreData
	DISPLAY prices USING currentStoreData
	DISPLAY descriptions USING currentStoreData
	DISPLAY statistics USING currentStoreData
END displayStore

BEGIN storeInit
	GET stores from storage
	IF stores does not exist
		CREATE placeholderStore data
		CREATE stores array containing placeholderStore
		SAVE stores into storage
	END IF
	GET currentStore from storage
	IF currentStore does not exist
		SET currentStore to "placeholderStore"
		SAVE currentStore into storage
	END IF
	GET stores from storage
	CONVERT stores into an accessible array
	SEARCH stores for currentStore
	IF currentStore is not found in stores
		CREATE placeholderStore data
		ADD placeholderStore to stores
		SAVE updated stores into storage
		SET currentStore to "placeholderStore"
		SAVE currentStore into storage
	END IF
	SEARCH stores for currentStore
	SET selectedStoreData to matching store
	RETURN selectedStoreData, currentStore
END storeInit
