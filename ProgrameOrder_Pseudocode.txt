BEGIN main
	SET currentStoreData to first item in CALL storeInit
	SET currentStore to second item in CALL storeInit
	CALL displayStore with data inputs of currentStore, and currentStoreData
	ONTRIGGER CALL addnewItem with data inputs of currentStore, and currentStoreData
END main

BEGIN displayStore
	DISPLAY restaurant name USING currentStore
	DISPLAY menu items USING currentStoreData
	DISPLAY prices USING currentStoreData
	DISPLAY descriptions USING currentStoreData
	DISPLAY statistics USING currentStoreData
END displayStore

BEGIN addnewItem
	PROMPT itemName
	IF itemName is in currentStoreData items data THEN
		DISPLAY Item already exists
		SET action to PROMPT want to modify(1)/remove(2) that time or Cancel action(3)
		IF action = modify or 1 THEN
			CALL modifyItem with data input of itemName, currentStoreData
		ELSE IF action = remove or 2 THEN
			CALL removeItem with data input of itemName
		END IF
		RETURN
	END IF
	SET editing to true
	WHILE editing = true
		PROMPT itemPrice
		PROMPT itemProductionCost
		PROMPT itemDescription
		PROMPT itemComplete
		IF itemComplete = y
			editing = false
		END IF
	END WHILE
	CREATE newItem WITH data
		name = itemName
		price = itemPrice
		productionCost = itemProductionCost
		description = itemDescription
	END CREATE
	ADD newItem currentStoreData items data
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage
	CALL displayStore with data inputs of currentStore, and currentStoreData
END addnewItem

BEGIN modifyItem
	SET itemData to GET itemName data from currentStoreData
	SET modifying to true
	WHILE modifying = true
		DISPLAY itemData
		PROMPT newItemPrice
		PROMPT newItemProductionCost
		PROMPT newItemDescription
		PROMPT modificationComplete
		IF modificationComplete = y
			modifying = false
		END IF
	END WHILE
	SET itemName itemData to
		price = newItemPrice
		productionCost = newItemProductionCost
		description = newItemDescription
	END SET
	SAVE updated itemName to currentStoreData item data
	SAVE updated currentStoreData to stores
	SAVE updated stores to storage
	CALL displayStore with data inputs of currentStore, and currentStoreData
END modifyItem

BEGIN removeItem
	PROMPT confirmRemoval
	IF confirmRemoval = y
		REMOVE itemName from currentStoreData
		SAVE updated currentStoreData to stores
		SAVE updated stores to storage
		CALL displayStore with data inputs of currentStore, and currentStoreData
	END IF
	RETURN
END removeItem

BEGIN storeInit
	GET stores from storage
	IF stores does not exist
		CREATE placeholderStore data
		CREATE stores array containing placeholderStore
		SAVE stores into storage
	END IF
	GET currentStore from storage
	IF currentStore does not exist
		SET currentStore to "placeholderStore"
		SAVE currentStore into storage
	END IF
	GET stores from storage
	CONVERT stores into an accessible array
	SEARCH stores for currentStore
	IF currentStore is not found in stores
		CREATE placeholderStore data
		ADD placeholderStore to stores
		SAVE updated stores into storage
		SET currentStore to "placeholderStore"
		SAVE currentStore into storage
	END IF
	SEARCH stores for currentStore
	SET selectedStoreData to matching store
	RETURN selectedStoreData, currentStore
END storeInit
